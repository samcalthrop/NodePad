## Parsing

- Record<string, any> to TreeNodeData:
```ts
export const convertDirectoryTree = (dirTree: DirectoryTree): TreeNodeData => {
  if (dirTree.children === undefined) {
    return {
      label: dirTree.name,
      value: dirTree.path,
    };
  }

  return {
    label: dirTree.name,
    value: dirTree.path,
    children: dirTree.children.map(convertDirectoryTree),
  };
};
```
